	--Transitions a little different in v4.

	Code for applying an ease to a transition:

				var t = d3.transition()
				    .duration(750)
				    .ease(d3.easeLinear);

				d3.selectAll(".apple").transition(t)
				    .style("fill", "red");

				d3.selectAll(".orange").transition(t)
				    .style("fill", "orange");
